# Erin Sub Synth Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a VST3 sub-oscillator synthesizer instrument plugin with C++/JUCE that loads in Ardour on Linux.

**Architecture:** Monolithic DSP with a small number of focused classes — Voice (oscillator + filter + envelopes), LFO (shared), PluginProcessor (voice management + parameter tree), PluginEditor (GUI), and a custom LookAndFeel (skeuomorphic rendering). CMake build system fetching JUCE via FetchContent.

**Tech Stack:** C++17, JUCE 7+, CMake, VST3 SDK (bundled with JUCE), Linux/GCC

#include "Voice.h"

void Voice::setSampleRate(double sr) { sampleRate = sr; }

float Voice::midiNoteToFreq(int note) const
{
    return 440.0f * std::pow(2.0f, (note - 69) / 12.0f);
}

void Voice::noteOn(int midiNote, float vel)
{
    currentNote = midiNote;
    velocity = vel;
    targetFrequency = midiNoteToFreq(midiNote + octaveOffset * 12);

    if (glideRate <= 0.0f || currentFrequency <= 0.0)
        currentFrequency = targetFrequency;

    phase1 = 0.0;
    phase2 = 0.0;
    ampStage = EnvStage::Attack;
    ampLevel = 0.0f;
    filterStage = EnvStage::Attack;
    filterEnvLevel = 0.0f;
    filterY1 = 0.0f;
    filterY2 = 0.0f;
}

void Voice::noteOff()
{
    ampStage = EnvStage::Release;
    filterStage = EnvStage::Release;
}

bool Voice::isActive() const
{
    return ampStage != EnvStage::Idle;
}

void Voice::setOctaveOffset(int offset) { octaveOffset = offset; }
void Voice::setOsc2Mix(float mix) { osc2Mix = mix; }
void Voice::setOsc2Detune(float cents) { osc2DetuneCents = cents; }
void Voice::setFilterCutoff(float hz) { filterCutoff = hz; }
void Voice::setFilterResonance(float res) { filterResonance = res; }
void Voice::setFilterEnvDepth(float depth) { filterEnvDepth = depth; }
void Voice::setGlideRate(float seconds) { glideRate = seconds; }

void Voice::setAmpADSR(float a, float d, float s, float r)
{
    ampAttack = a; ampDecay = d; ampSustain = s; ampRelease = r;
}

void Voice::setFilterADSR(float a, float d, float s, float r)
{
    filtAttack = a; filtDecay = d; filtSustain = s; filtRelease = r;
}

void Voice::setTargetNote(int midiNote)
{
    targetFrequency = midiNoteToFreq(midiNote + octaveOffset * 12);
    currentNote = midiNote;
}

float Voice::processEnvelope(EnvStage& stage, float& level, float a, float d, float s, float r)
{
    float rate;
    switch (stage)
    {
        case EnvStage::Attack:
            rate = (a > 0.0f) ? 1.0f / (a * (float)sampleRate) : 1.0f;
            level += rate;
            if (level >= 1.0f) { level = 1.0f; stage = EnvStage::Decay; }
            break;
        case EnvStage::Decay:
            rate = (d > 0.0f) ? 1.0f / (d * (float)sampleRate) : 1.0f;
            level -= rate * (1.0f - s);
            if (level <= s) { level = s; stage = EnvStage::Sustain; }
            break;
        case EnvStage::Sustain:
            level = s;
            break;
        case EnvStage::Release:
            rate = (r > 0.0f) ? 1.0f / (r * (float)sampleRate) : 1.0f;
            level -= rate;
            if (level <= 0.0f) { level = 0.0f; stage = EnvStage::Idle; }
            break;
        case EnvStage::Idle:
            level = 0.0f;
            break;
    }
    return level;
}

void Voice::updateFilter(float cutoffHz)
{
    // Clamp cutoff to safe range
    cutoffHz = std::max(20.0f, std::min(cutoffHz, (float)(sampleRate * 0.45)));
    float w = std::tan(juce::MathConstants<float>::pi * cutoffHz / (float)sampleRate);
    float q = 0.5f + filterResonance * 9.5f; // Q range: 0.5 to 10
    float norm = 1.0f / (w * w + w / q + 1.0f);
    // Coefficients stored implicitly — apply 2-pole SVF-style in processSample
    // Using simple 2-pole IIR for clarity
    (void)norm; // filter applied directly in processSample
}

float Voice::processSample(float lfoFilterMod, float lfoPitchMod)
{
    if (!isActive()) return 0.0f;

    // Glide
    if (glideRate > 0.0f && currentFrequency != targetFrequency)
    {
        double glideStep = 1.0 / (glideRate * sampleRate);
        if (currentFrequency < targetFrequency)
            currentFrequency = std::min(currentFrequency * (1.0 + glideStep), targetFrequency);
        else
            currentFrequency = std::max(currentFrequency * (1.0 - glideStep), targetFrequency);
    }

    // Apply LFO pitch modulation (in cents)
    double freq = currentFrequency * std::pow(2.0, lfoPitchMod / 1200.0);

    // Oscillator 1
    double inc1 = freq / sampleRate;
    phase1 += inc1;
    if (phase1 >= 1.0) phase1 -= 1.0;
    float osc1 = std::sin(2.0 * juce::MathConstants<double>::pi * phase1);

    // Oscillator 2 (detuned)
    double freq2 = freq * std::pow(2.0, osc2DetuneCents / 1200.0);
    double inc2 = freq2 / sampleRate;
    phase2 += inc2;
    if (phase2 >= 1.0) phase2 -= 1.0;
    float osc2 = std::sin(2.0 * juce::MathConstants<double>::pi * phase2);

    // Mix oscillators
    float sample = osc1 * (1.0f - osc2Mix) + osc2 * osc2Mix;

    // Process envelopes
    float ampEnv = processEnvelope(ampStage, ampLevel, ampAttack, ampDecay, ampSustain, ampRelease);
    float filtEnv = processEnvelope(filterStage, filterEnvLevel, filtAttack, filtDecay, filtSustain, filtRelease);

    // Calculate effective filter cutoff
    float envMod = filtEnv * filterEnvDepth * 2000.0f;
    float lfoMod = lfoFilterMod * 2000.0f;
    float effectiveCutoff = std::max(20.0f, std::min(filterCutoff + envMod + lfoMod, 20000.0f));

    // Simple 2-pole low-pass filter (resonant)
    float w = std::tan(juce::MathConstants<float>::pi * effectiveCutoff / (float)sampleRate);
    float q = 0.707f + filterResonance * 9.3f;
    float g = w / (1.0f + w / q + w * w);
    float hp = (sample - (1.0f / q + w) * filterY1 - filterY2) / (1.0f + w / q + w * w);
    float bp = w * hp + filterY1;
    float lp = w * bp + filterY2;
    filterY1 = w * hp + bp;
    filterY2 = w * bp + lp;

    // Apply amp envelope and velocity
    return lp * ampEnv * velocity;
}

#include "LFO.h"

void LFO::setSampleRate(double sr) { sampleRate = sr; }
void LFO::setRate(float hz) { rate = hz; }
void LFO::setWaveform(Waveform w) { waveform = w; }
void LFO::setPitchDepth(float cents) { pitchDepth = cents; }
void LFO::setFilterDepth(float depth) { filterDepth = depth; }

void LFO::reset()
{
    phase = 0.0;
    currentValue = 0.0f;
}

void LFO::advance()
{
    phase += rate / sampleRate;
    if (phase >= 1.0) phase -= 1.0;

    switch (waveform)
    {
        case Waveform::Sine:
            currentValue = std::sin(2.0 * juce::MathConstants<double>::pi * phase);
            break;
        case Waveform::Triangle:
            currentValue = (float)(4.0 * std::abs(phase - 0.5) - 1.0);
            break;
        case Waveform::Square:
            currentValue = (phase < 0.5) ? 1.0f : -1.0f;
            break;
    }
}

float LFO::getPitchMod() const
{
    return currentValue * pitchDepth;
}

float LFO::getFilterMod() const
{
    return currentValue * filterDepth;
}

void ErinSubSynthProcessor::updateVoiceParameters()
{
    int octaveIndex = apvts.getRawParameterValue("octave")->load();
    int octaveOffset = octaveIndex - 2; // index 0=-2, 1=-1, 2=0

    float osc2Mix = apvts.getRawParameterValue("osc2mix")->load();

    float cutoff = apvts.getRawParameterValue("cutoff")->load();
    float resonance = apvts.getRawParameterValue("resonance")->load();

    float aA = apvts.getRawParameterValue("ampAttack")->load();
    float aD = apvts.getRawParameterValue("ampDecay")->load();
    float aS = apvts.getRawParameterValue("ampSustain")->load();
    float aR = apvts.getRawParameterValue("ampRelease")->load();

    float fA = apvts.getRawParameterValue("filtAttack")->load();
    float fD = apvts.getRawParameterValue("filtDecay")->load();
    float fS = apvts.getRawParameterValue("filtSustain")->load();
    float fR = apvts.getRawParameterValue("filtRelease")->load();
    float fDepth = apvts.getRawParameterValue("filtEnvDepth")->load();

    float glide = apvts.getRawParameterValue("glide")->load();

    for (auto& v : voices)
    {
        v.setOctaveOffset(octaveOffset);
        v.setOsc2Mix(osc2Mix);
        v.setOsc2Detune(osc2Detune);
        v.setFilterCutoff(cutoff);
        v.setFilterResonance(resonance);
        v.setAmpADSR(aA, aD, aS, aR);
        v.setFilterADSR(fA, fD, fS, fR);
        v.setFilterEnvDepth(fDepth);
        v.setGlideRate(glide);
    }

    // LFO params
    lfo.setRate(apvts.getRawParameterValue("lfoRate")->load());
    lfo.setWaveform(static_cast<LFO::Waveform>(
        (int)apvts.getRawParameterValue("lfoWaveform")->load()));
    lfo.setPitchDepth(apvts.getRawParameterValue("lfoPitchDepth")->load());
    lfo.setFilterDepth(apvts.getRawParameterValue("lfoFilterDepth")->load());
}

void ErinSubSynthProcessor::handleNoteOn(int midiNote, float vel)
{
    bool mono = ((int)apvts.getRawParameterValue("voiceMode")->load()) == 0;

    if (mono)
    {
        voices[0].setTargetNote(midiNote);
        if (!voices[0].isActive())
            voices[0].noteOn(midiNote, vel);
        voices[0].setNoteOrder(++noteCounter);
    }
    else
    {
        // Find free voice or steal oldest
        int freeIdx = -1;
        int oldestIdx = 0;
        int oldestOrder = INT_MAX;

        for (int i = 0; i < maxVoices; ++i)
        {
            if (!voices[i].isActive() && freeIdx < 0)
                freeIdx = i;
            if (voices[i].getNoteOrder() < oldestOrder)
            {
                oldestOrder = voices[i].getNoteOrder();
                oldestIdx = i;
            }
        }

        int idx = (freeIdx >= 0) ? freeIdx : oldestIdx;
        voices[idx].noteOn(midiNote, vel);
        voices[idx].setNoteOrder(++noteCounter);
    }
}

void ErinSubSynthProcessor::handleNoteOff(int midiNote)
{
    for (auto& v : voices)
        if (v.getCurrentNote() == midiNote && v.isActive())
            v.noteOff();
}

void ErinSubSynthProcessor::processBlock(juce::AudioBuffer<float>& buffer, juce::MidiBuffer& midi)
{
    buffer.clear();
    updateVoiceParameters();

    float masterGain = juce::Decibels::decibelsToGain(
        apvts.getRawParameterValue("masterVol")->load(), -60.0f);

    auto* leftChannel = buffer.getWritePointer(0);
    auto* rightChannel = buffer.getNumChannels() > 1 ? buffer.getWritePointer(1) : nullptr;

    int samplePos = 0;

    for (const auto metadata : midi)
    {
        auto msg = metadata.getMessage();
        int msgPos = metadata.samplePosition;

        // Render audio up to this MIDI event
        for (; samplePos < msgPos; ++samplePos)
        {
            lfo.advance();
            float pitchMod = lfo.getPitchMod();
            float filterMod = lfo.getFilterMod();

            float sample = 0.0f;
            for (auto& v : voices)
                sample += v.processSample(filterMod, pitchMod);

            sample *= masterGain;
            leftChannel[samplePos] = sample;
            if (rightChannel) rightChannel[samplePos] = sample;
        }

        // Handle MIDI event
        if (msg.isNoteOn())
            handleNoteOn(msg.getNoteNumber(), msg.getFloatVelocity());
        else if (msg.isNoteOff())
            handleNoteOff(msg.getNoteNumber());
    }

    // Render remaining samples
    for (; samplePos < buffer.getNumSamples(); ++samplePos)
    {
        lfo.advance();
        float pitchMod = lfo.getPitchMod();
        float filterMod = lfo.getFilterMod();

        float sample = 0.0f;
        for (auto& v : voices)
            sample += v.processSample(filterMod, pitchMod);

        sample *= masterGain;
        leftChannel[samplePos] = sample;
        if (rightChannel) rightChannel[samplePos] = sample;
    }
}

void ErinSubSynthProcessor::getStateInformation(juce::MemoryBlock& destData)
{
    auto state = apvts.copyState();
    std::unique_ptr<juce::XmlElement> xml(state.createXml());
    copyXmlToBinary(*xml, destData);
}

void ErinSubSynthProcessor::setStateInformation(const void* data, int sizeInBytes)
{
    std::unique_ptr<juce::XmlElement> xml(getXmlFromBinary(data, sizeInBytes));
    if (xml && xml->hasTagName(apvts.state.getType()))
        apvts.replaceState(juce::ValueTree::fromXml(*xml));
}

}

void ErinLookAndFeel::drawRotarySlider(juce::Graphics& g, int x, int y,
    int width, int height, float sliderPos,
    float rotaryStartAngle, float rotaryEndAngle, juce::Slider&)
{
    auto bounds = juce::Rectangle<int>(x, y, width, height).toFloat().reduced(4.0f);
    auto radius = std::min(bounds.getWidth(), bounds.getHeight()) / 2.0f;
    auto centreX = bounds.getCentreX();
    auto centreY = bounds.getCentreY();
    auto angle = rotaryStartAngle + sliderPos * (rotaryEndAngle - rotaryStartAngle);

    // Knob shadow
    g.setColour(juce::Colours::black.withAlpha(0.5f));
    g.fillEllipse(centreX - radius + 2, centreY - radius + 2, radius * 2, radius * 2);

    // Knob body — gradient for 3D effect
    juce::ColourGradient bodyGrad(knobBody.brighter(0.3f), centreX, centreY - radius,
                                   knobBody.darker(0.3f), centreX, centreY + radius, false);
    g.setGradientFill(bodyGrad);
    g.fillEllipse(centreX - radius, centreY - radius, radius * 2, radius * 2);

    // Knob edge ring
    g.setColour(knobBody.darker(0.5f));
    g.drawEllipse(centreX - radius, centreY - radius, radius * 2, radius * 2, 1.5f);

    // Indicator line
    juce::Path indicator;
    auto indicatorLength = radius * 0.7f;
    auto lineW = 2.5f;
    indicator.addRectangle(-lineW * 0.5f, -indicatorLength, lineW, indicatorLength);
    indicator.applyTransform(juce::AffineTransform::rotation(angle).translated(centreX, centreY));
    g.setColour(knobIndicator);
    g.fillPath(indicator);

    // Center cap
    auto capRadius = radius * 0.2f;
    g.setColour(knobBody.brighter(0.1f));
    g.fillEllipse(centreX - capRadius, centreY - capRadius, capRadius * 2, capRadius * 2);
}

#include "Voice.h"
#include "LFO.h"
#include <cassert>
#include <cmath>
#include <iostream>

constexpr double TEST_SR = 44100.0;
constexpr float TOLERANCE = 0.01f;

bool approxEqual(float a, float b, float tol = TOLERANCE)
{
    return std::abs(a - b) < tol;
}

void testOscillatorFrequency()
{
    std::cout << "Test: oscillator produces correct frequency... ";

    Voice v;
    v.setSampleRate(TEST_SR);
    v.setOctaveOffset(0);
    v.setOsc2Mix(0.0f);
    v.setFilterCutoff(20000.0f);  // open filter wide
    v.setFilterResonance(0.0f);
    v.setAmpADSR(0.0f, 0.0f, 1.0f, 5.0f);  // instant attack, full sustain
    v.setFilterADSR(0.0f, 0.0f, 1.0f, 5.0f);
    v.setFilterEnvDepth(0.0f);

    // Play A4 (MIDI 69 = 440 Hz)
    v.noteOn(69, 1.0f);

    // Count zero crossings over 1 second to estimate frequency
    int zeroCrossings = 0;
    float prevSample = 0.0f;
    int numSamples = (int)TEST_SR;

    for (int i = 0; i < numSamples; ++i)
    {
        float s = v.processSample(0.0f, 0.0f);
        if (i > 0 && ((prevSample >= 0.0f && s < 0.0f) || (prevSample < 0.0f && s >= 0.0f)))
            zeroCrossings++;
        prevSample = s;
    }

    // Sine wave has 2 zero crossings per cycle
    float estimatedFreq = zeroCrossings / 2.0f;
    assert(std::abs(estimatedFreq - 440.0f) < 5.0f);

    std::cout << "PASS (estimated " << estimatedFreq << " Hz)" << std::endl;
}

void testOctaveOffset()
{
    std::cout << "Test: octave offset shifts frequency correctly... ";

    Voice v;

    {

    assert(std::abs(estimatedFreq - 220.0f) < 5.0f);

}

void testEnvelopeADSR()
{
    std::cout << "Test: amplitude envelope follows ADSR stages... ";

    Voice v;
    v.setSampleRate(TEST_SR);
    v.setOctaveOffset(0);
    v.setOsc2Mix(0.0f);
    v.setFilterCutoff(20000.0f);
    v.setFilterResonance(0.0f);
    v.setFilterADSR(0.0f, 0.0f, 1.0f, 5.0f);
    v.setFilterEnvDepth(0.0f);

    // 100ms attack, 100ms decay, 0.5 sustain, 100ms release
    v.setAmpADSR(0.1f, 0.1f, 0.5f, 0.1f);
    v.noteOn(69, 1.0f);

    // After attack phase (100ms = 4410 samples), output should be near peak
    for (int i = 0; i < 4410; ++i)
        v.processSample(0.0f, 0.0f);

    assert(v.isActive());

    // After sustain, trigger release
    for (int i = 0; i < 4410; ++i)
        v.processSample(0.0f, 0.0f);

    v.noteOff();

    // After release (100ms), voice should be inactive
    for (int i = 0; i < 5000; ++i)
        v.processSample(0.0f, 0.0f);

    assert(!v.isActive());

    std::cout << "PASS" << std::endl;
}

void testLFOWaveforms()
{
    std::cout << "Test: LFO produces correct waveforms... ";

    LFO lfo;
    lfo.setSampleRate(TEST_SR);
    lfo.setRate(1.0f);  // 1 Hz
    lfo.setPitchDepth(100.0f);
    lfo.setFilterDepth(1.0f);

    // Test sine: after 1/4 cycle (quarter period) should be near +1
    lfo.setWaveform(LFO::Waveform::Sine);
    lfo.reset();
    int quarterCycle = (int)(TEST_SR / 4.0);
    for (int i = 0; i < quarterCycle; ++i)
        lfo.advance();

    float pitchMod = lfo.getPitchMod();
    assert(std::abs(pitchMod - 100.0f) < 2.0f);  // near max depth

    // Test square: should be +1 for first half, -1 for second half
    lfo.setWaveform(LFO::Waveform::Square);
    lfo.reset();
    lfo.advance();
    assert(lfo.getFilterMod() > 0.9f);  // first sample of first half

    std::cout << "PASS" << std::endl;
}

void testVoiceAllocation()
{
    std::cout << "Test: voice is active after noteOn, inactive after release... ";

    Voice v;
    v.setSampleRate(TEST_SR);
    v.setOctaveOffset(0);
    v.setAmpADSR(0.0f, 0.0f, 1.0f, 0.001f);

    assert(!v.isActive());

    std::cout << "PASS" << std::endl;
}

}
